<script>
// ===== DATA =====
let tasks = JSON.parse(localStorage.getItem('mahaam_tasks') || 'null') || [
  {id:1,text:"مراجعة تقرير المشروع",cat:"work",pri:"high",done:false,createdAt:Date.now(),deadline:new Date(Date.now()+1000*60*60*3).toISOString().slice(0,16)},
  {id:2,text:"ممارسة الرياضة",cat:"health",pri:"medium",done:false,createdAt:Date.now(),deadline:new Date(Date.now()+1000*60*60*26).toISOString().slice(0,16)},
  {id:3,text:"قراءة فصل من الكتاب",cat:"learning",pri:"low",done:true,createdAt:Date.now(),deadline:""},
];

let selectedCat = 'personal';
let selectedPri = 'medium';
let filterCat = 'all';
let showDone = true;
let formOpen = false;

const CAT_COLORS = {work:'#60a5fa',personal:'#f472b6',health:'#34d399',learning:'#a78bfa'};
const PRI_COLORS = {high:'#f87171',medium:'#fbbf24',low:'#34d399'};
const CAT_LABELS = {work:'عمل',personal:'شخصي',health:'صحة',learning:'تعلم'};
const PRI_LABELS = {high:'عالية',medium:'متوسطة',low:'منخفضة'};

function save(){localStorage.setItem('mahaam_tasks',JSON.stringify(tasks));}

// ===== TIME =====
function getTimeLeft(deadline){
  if(!deadline) return null;
  const diff = new Date(deadline) - new Date();
  if(diff<=0) return {label:'انتهى الوقت!',urgent:true,overdue:true,cls:'badge-overdue',icon:'🔴'};
  const days=Math.floor(diff/86400000), hours=Math.floor((diff%86400000)/3600000), mins=Math.floor((diff%3600000)/60000);
  if(days>0) return {label:`${days} يوم ${hours} ساعة`,urgent:days<=1,overdue:false,cls:days<=1?'badge-urgent':'badge-ok',icon:days<=1?'🟡':'🟢'};
  if(hours>0) return {label:`${hours} ساعة ${mins} دقيقة`,urgent:true,overdue:false,cls:'badge-urgent',icon:'🟡'};
  return {label:`${mins} دقيقة`,urgent:true,overdue:false,cls:'badge-urgent',icon:'🟡'};
}

function formatDate(dl){
  return new Date(dl).toLocaleString('ar-SA',{month:'short',day:'numeric',hour:'2-digit',minute:'2-digit'});
}

// ===== FORM =====
function toggleForm(){
  formOpen=!formOpen;
  document.getElementById('formBox').className='form-box'+(formOpen?' visible':'');
  const btn=document.getElementById('toggleFormBtn');
  btn.textContent=formOpen?'✕ إلغاء':'+ مهمة جديدة';
  btn.className='btn-add'+(formOpen?' open':'');
  if(formOpen) setTimeout(()=>document.getElementById('taskInput').focus(),100);
}

function selectCat(c){
  selectedCat=c;
  document.querySelectorAll('#catChips .chip').forEach(b=>{
    b.className='chip'+(b.dataset.cat===c?' active-cat-'+c:'');
  });
}

function selectPri(p){
  selectedPri=p;
  document.querySelectorAll('#priChips .chip').forEach(b=>{
    b.className='chip'+(b.dataset.pri===p?' active-pri-'+p:'');
  });
}

function onDeadlineChange(){
  const val=document.getElementById('deadlineInput').value;
  const input=document.getElementById('deadlineInput');
  const preview=document.getElementById('deadlinePreview');
  if(val){
    input.className='deadline-input has-value';
    const tl=getTimeLeft(val);
    document.getElementById('deadlineTimeLeft').textContent=tl?tl.label:'';
    preview.className='deadline-preview visible';
  } else {
    input.className='deadline-input';
    preview.className='deadline-preview';
  }
}

function addTask(){
  const text=document.getElementById('taskInput').value.trim();
  if(!text) return;
  const deadline=document.getElementById('deadlineInput').value;
  tasks.unshift({id:Date.now(),text,cat:selectedCat,pri:selectedPri,done:false,createdAt:Date.now(),deadline});
  save();
  document.getElementById('taskInput').value='';
  document.getElementById('deadlineInput').value='';
  document.getElementById('deadlineInput').className='deadline-input';
  document.getElementById('deadlinePreview').className='deadline-preview';
  toggleForm();
  render();
}

// ===== FILTER =====
function setFilter(f){
  filterCat=f;
  document.querySelectorAll('.filter-chip').forEach(b=>{
    b.className='filter-chip '+b.dataset.filter+(b.dataset.filter===f?' active':'');
  });
  render();
}

function toggleDone(){
  showDone=!showDone;
  const btn=document.getElementById('toggleDoneBtn');
  btn.textContent=showDone?'إخفاء المنجزة':'إظهار المنجزة';
  btn.className='toggle-done'+(showDone?'':' hidden-mode');
  render();
}

// ===== ACTIONS =====
function toggleTask(id){
  tasks=tasks.map(t=>t.id===id?{...t,done:!t.done}:t);
  save(); render();
}

function deleteTask(id){
  tasks=tasks.filter(t=>t.id!==id);
  save(); render();
}

// ===== RENDER =====
function render(){
  const pending=tasks.filter(t=>!t.done).length;
  const total=tasks.length;
  const done=total-pending;
  const pct=total===0?0:Math.round((done/total)*100);
  const overdue=tasks.filter(t=>!t.done&&t.deadline&&new Date(t.deadline)<new Date()).length;

  document.getElementById('progressFill').style.width=pct+'%';
  document.getElementById('progressPct').textContent=pct+'%';
  document.getElementById('subtext').innerHTML=`${pending} متبقية${overdue>0?` <span class="overdue-badge">● ${overdue} متأخرة</span>`:''}`;

  const statsBox=document.getElementById('statsBox');
  statsBox.style.display=total>0?'grid':'none';
  document.getElementById('statTotal').textContent=total;
  document.getElementById('statDone').textContent=done;
  document.getElementById('statPending').textContent=pending;

  const filtered=tasks.filter(t=>{
    const catOk=filterCat==='all'||t.cat===filterCat;
    const doneOk=showDone||!t.done;
    return catOk&&doneOk;
  }).sort((a,b)=>{
    if(a.done!==b.done) return a.done?1:-1;
    if(a.deadline&&b.deadline) return new Date(a.deadline)-new Date(b.deadline);
    if(a.deadline) return -1;
    if(b.deadline) return 1;
    return b.createdAt-a.createdAt;
  });

  const list=document.getElementById('tasksList');

  if(filtered.length===0){
    list.innerHTML=`<div class="empty"><div class="empty-icon">✓</div><div class="empty-text">لا توجد مهام</div></div>`;
    return;
  }

  list.innerHTML=filtered.map((t,i)=>{
    const cc=CAT_COLORS[t.cat]||'#60a5fa';
    const pc=PRI_COLORS[t.pri]||'#fbbf24';
    const tl=t.deadline?getTimeLeft(t.deadline):null;
    const cardBg=t.done?'rgba(15,23,42,0.4)':tl?.overdue?'rgba(248,113,113,0.06)':'rgba(30,27,75,0.6)';
    const cardBorder=t.done?'rgba(255,255,255,0.03)':tl?.overdue?'rgba(248,113,113,0.25)':tl?.urgent?'rgba(251,191,36,0.2)':'rgba(255,255,255,0.06)';

    const deadlineBadge = t.deadline&&!t.done ? `
      <div class="deadline-badge ${tl?.cls||'badge-ok'}">
        <span style="font-size:16px">${tl?.icon||'🟢'}</span>
        <div>
          <div class="time-label">${tl?.label||''}</div>
          <div class="time-date">${formatDate(t.deadline)}</div>
        </div>
      </div>` : '';

    return `<div class="task-card${t.done?' done-card':''}" style="background:${cardBg};border-color:${cardBorder};border-right-color:${t.done?'rgba(255,255,255,0.04)':cc};animation-delay:${i*0.05}s">
      <div class="check-circle" onclick="toggleTask(${t.id})" style="border-color:${t.done?cc:'rgba(255,255,255,0.15)'};background:${t.done?cc+'33':'transparent'};color:${cc}">
        ${t.done?'✓':''}
      </div>
      <div class="task-content">
        <div class="task-text${t.done?' done-text':''}" style="color:${t.done?'#334155':'#e2e8f0'}">${t.text}</div>
        ${deadlineBadge}
        <div class="tags-row">
          <span class="tag" style="color:${cc}cc;background:${cc}11;border-color:${cc}22">${CAT_LABELS[t.cat]||t.cat}</span>
